<h1>Mencari Huruf</h1>

```
Buatlah program untuk mencari kemunculan pertama suatu huruf dari teks yang diberikan. Keluaran adalah posisi huruf relatif terhadap awal teks. Jika huruf tidak ditemukan, cetak tidak ada. Panjang teks selalu 50 karakter.
```

Contoh Masukan

```
satuduatigaempatlimaenamtujuhdelapansembilansepulu
d
```

Contoh Keluaran

```
5
```

<h2> Code </h2>

```
extern printf
extern scanf
extern fflush
 ;22. panjang kata (asm)
  
  
     
section .data
    input       db  "%s %s", 0
    output      db  "%d", 10, 0
    no         db  "tidak ada", 10, 0
  
section .bss
    huruf      resb    50
    cek         resb    2
  
section .text
 global main
main:
  
    push cek ;scanf dan di masukin ke sting
    push huruf ; yang di scan cek duu
    push input
    call scanf
    add esp, 12
     
       ; normalin, ngeclear
 
    mov al, [cek] ; yang di cari
    mov edi, huruf ; string hurufnya
    mov ecx, 50
    repne scasb    ;scan katanya pertehur, teru sscan sampai ketemu null
     
 
    mov eax, 50
    sub eax, ecx
    mov ebx, 0
    cmp ecx, ebx
    je tidak
 
    push eax        ; untuk ngeprint, letak null
    push output     ; masukin ke output
    call printf
    add esp, 8
     
    jmp exit
     
tidak:
    push no
    call printf
    add esp, 4
 
 
;keluar
exit:
    push 0
    call fflush
      
    mov eax, 1
    mov ebx, 0
    int 80h
```
